ENT0	BIT	P3.2
ENT1	BIT	P3.3	
	ORG	0
	LJMP	MAIN

	ORG	0003H
	LJMP	EST0
	
	ORG	0013H
	LJMP	EST1
	
EST0:
	JMP	LOOP1

EST1:
	JMP	LOOP2

MAIN:	
	MOV	P1,#0
	MOV	A,#00010001B
	SETB	EX0
	SETB	EX1
	SETB	EA
	JMP	$

LOOP1:	
	RL	A
	LCALL	DELAY
	MOV	P1,A
	LCALL	DELAY
	JB	ENT1,LOOP1
	RETI
LOOP2:
	RR	A
	LCALL	DELAY
	MOV	P1,A
	LCALL	DELAY	
	JB	ENT0,LOOP2
	RETI

DELAY:
	MOV     R7,#0FFH

L1:
	MOV    R6,#0FFH

L2:
	DJNZ    R6,L2
	DJNZ    R7,L1
	RET

	END
